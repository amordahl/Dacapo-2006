17      public Object visit(ASTMethodDeclarator node, Object data) {
18          Map params = node.getScope().getVariableDeclarations();
19          for (Iterator i = params.keySet().iterator(); i.hasNext();) {
20              VariableNameDeclaration decl = (VariableNameDeclaration) i.next();
21              List usages = (List) params.get(decl);
22              for (Iterator j = usages.iterator(); j.hasNext();) {
23                  NameOccurrence occ = (NameOccurrence) j.next();
24                  if ((occ.isOnLeftHandSide() || occ.isSelfAssignment()) && occ.getNameForWhichThisIsAQualifier() == null && !decl.isArray()) {
25                      addViolation(data, decl.getNode(), decl.getImage());
26                  }
27              }
28          }
